#gatekeeper
planet_event = {
	id = chrysalis_mod.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				NOT = {
					has_planet_flag = chrysalis_mod_chrysalis_capital
				}
			}
			set_planet_flag = chrysalis_mod_chrysalis_capital
			planet_event = { id = chrysalis_mod.2 days = 25000 random = 5000}
		}
		random_list = {
			40 = {
				#fallen empire
				if = { 
					limit = { 
						any_country = { 
							is_country_type = fallen_empire
						}
					}
					random_country = {
						limit = {
							is_country_type = fallen_empire
						}
						set_country_flag = chrysalis_mod_ruiner
					}
				}
					else = {
						planet_event = { id = chrysalis_mod.1 }
					}
			}
			40 = {
				#nearby empire
				if = { 
					limit = { 
						any_country = { 
							is_ai = yes
							NOT = { is_country_type = fallen_empire }									
						}
					}
				
					}
					random_country = {
						limit = {							
								is_ai = yes
								NOT = { is_country_type = fallen_empire }							
						}
						set_country_flag = chrysalis_mod_ruiner
					}
					else = {
						planet_event = { id = chrysalis_mod.1 }
					}
				
			}
			20 = {
				set_planet_flag = chrysalis_mod_ruiner_precursor
			}
		}
	}
}

country_event = {
	id = chrysalis_mod.3
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		save_event_target_as = current_chrysallis
		random_country = {
			limit = {
				has_country_flag = chrysalis_mod_ruiner
				NOT = {
					has_country_flag = chrysalis_mod_taken
				}
			}
			set_relation_flag = {
				who = event_target:current_chrysallis
				flag = relation_flag_chrysalis_toward_ruiner
			}
			set_country_flag = chrysalis_mod_taken
		}

		# setting flags
		set_country_flag = chrysalis_country

		# setting graphical culture
		set_graphical_culture = drone

		# baseline resource stockpile
		add_resource = { energy = 5000 }
		add_resource = { minerals = 3000 }
		add_resource = { influence = 500 }

		# flip capital planet and add at least 5 Machine pops
		from = {
			# flip military stations
			set_owner = root
			set_controller = root
			set_capital = yes

			solar_system = {
				if = {
					limit = {
						exists = starbase
					}
					starbase = { set_owner = root }
				}
					else = {
						create_starbase = {
							size = starbase_starport
							module = shipyard
							owner = root
						}
					}				
			}

			generate_start_deposits_and_blockers =  yes
			generate_start_buildings_and_districts = yes

			while = {
				limit = { free_housing  > 0 }
					create_pop = {
						species = root
					}
			}
			
			# create armies
			while = {
				count = 3
				create_army = {
					owner = root
					type = machine_assault_1
				}
			}
		}

		# create fleets
		create_fleet = {
			effect = {
				set_owner = root
				create_ship = {
					name = "chrysalis_mod_vengeance"
					random_existing_design = titan
				}
				set_fleet_stance = passive
				set_location = root.capital_scope
			}
		}

		#science
		create_fleet = {
			effect = {
				set_owner = root

				create_ship = {
					name = random
					random_existing_design = science
				}

				set_fleet_stance = evasive				
				set_location = root.capital_scope
				
				owner = {
					create_leader = {
						class = scientist
						sub_type = survey
						name = random
						species = owner_main_species
					}
				}
				set_leader = last_created_leader
			}			
		}
			
		#constructor
		create_fleet = {
			effect = {
				set_owner = root
				
				create_ship = {
					name = random
					random_existing_design = constructor
				}
				
				set_fleet_stance = evasive
				set_location = root.capital_scope
			}
		}
	}
}

country_event = {
	id = chrysalis_mod.5
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		save_event_target_as = current_chrysallis
		from.owner = { save_event_target_as = chrysalis_mod_old }

		# setting flags
		set_country_flag = chrysalis_country_new
		every_country = {
			limit = { has_communications = event_target:chrysalis_mod_old}
			establish_communications_no_message = root
			establish_contact = { who = root location = from }
		}
		every_country = {
			limit = {
				has_communications = event_target:chrysalis_mod_old
				is_ai = no
			}
			country_event = { 
				id = chrysalis_mod.2006
			}
		}
		
		# setting graphical culture
		set_graphical_culture = event_target:chrysalis_mod_old

		# baseline resource stockpile: set to old values
		add_resource = { energy = 5000 }
		add_resource = { minerals = 3000 }
		add_resource = { influence = 500 }

		event_target:chrysalis_mod_old = {
			capital_scope = {
				save_event_target_as = chrysalis_mod_old_capital
			}

			every_owned_planet = {
				# flip military stations
				set_owner = root
				set_controller = root

				solar_system = {
					if = {
						limit = {
							exists = starbase
						}
						starbase = { set_owner = root }
					}					
				}

				every_owned_pop = {
					if = {
						limit = {
							has_trait = "trait_machine_unit"
						}
						random_list = {
							50 = {
									remove_trait = "trait_machine_unit"
									add_trait = "trait_robotic_3"								
							}
							50 = {
								kill_pop = yes
								prev = {
										create_pop = {
											species = root
										}
								}
							}
						}
					}
				}
			}

			every_owned_army = {
				set_owner = root
			}

			every_owned_fleet = {
				set_owner = root
				set_controller = root
			}		
		}
		event_target:chrysalis_mod_old_capital = {
			set_capital = yes
		}

	}
}

country_event = {
	id = chrysalis_mod.6
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		random_planet = {
			limit = {
				is_planet_class = pc_nuked
				is_capital = no
				OR = {
					AND = {
						exists = owner
						owner = {
							num_owned_planets > 3
						}
					}
					NOT = {
						exists = owner
					}
				}
				planet_size > 14
			}
		}
	}
}
